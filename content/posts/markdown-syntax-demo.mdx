### Code Block With Language

```ts
export const LANGUAGES = ['en', 'zh'] as const
export type Language = (typeof LANGUAGES)[number]

export function getI18nOptions(
  lng: Language = 'en',
  ns: string | string[] = 'common'
) {
  return {
    supportedLngs: LANGUAGES,
    fallbackLng: 'en',
    lng,
    ns
  }
}
```
